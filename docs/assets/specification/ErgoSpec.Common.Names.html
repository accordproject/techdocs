Section Debug.
  Definition DEBUG_aux {A} s (d1 d2 : A) : A
    := match LOG_string s with
       | y => if unit_eqdec y tt then d1 else d2
       end.
  Definition DEBUG {A} s (d:A) : A :=
    DEBUG_aux s d d.
                                   
End Debug.
