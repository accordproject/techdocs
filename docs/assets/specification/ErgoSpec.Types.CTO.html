    Definition cto_annot (ct:cto_type) : A :=
      match ct with
      | CTOBoolean a => a
      | CTOString a => a
      | CTODouble a => a
      | CTOLong a => a
      | CTOInteger a => a
      | CTODateTime a => a
      | CTOClassRef a _ => a
      | CTOOption a _ => a
      | CTOArray a _ => a
      end.
